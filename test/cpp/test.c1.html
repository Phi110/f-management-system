    #include <string>
    #include <vector>
    using namespace std;
    
    class Takoyaki {
    public:
        string name;
        string where;
        int price;
        Takoyaki() {
            name = "ソース";
            where = "店内";
            price = 800;
        }
        Takoyaki(string name0, int price0) {
            name = name0;
            where = "お持ち帰り";
            price = price0;
        }
        Takoyaki(string name0, string where0, int price0) {
            name = name0;
            where = where0;
            price = price0;
        }
        Takoyaki add(Takoyaki t) {
            string name0 = name + "と" + t.name;
            string where0;
            if (where == t.where) {
                where0 = where;
            } else {
                where0 = "店内";
            }
            int price0 = price + t.price;
            Takoyaki t0 {name0, where0, price0};
            return t0;
        }
        string checkout() {
            if (where == "お持ち帰り") {
                price *= 0.98;
            }
            return name + "を"
             + to_string(price) + "円分買います\n";
        }
    };
    
    int main() {
        Takoyaki t0 {"てりたま", "お持ち帰り", 1000};
        Takoyaki t1 {"ねぎだこ", 900};
        vector<Takoyaki> fune {t0, t1};
        Takoyaki t {};
        for (int i = 0; i < fune.size(); i++) {
            t = t.add(fune[i]);
        }
        cout << t.checkout() << endl;
